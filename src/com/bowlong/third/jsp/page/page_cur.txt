<script type="text/javascript">
	function click2sendPaging(curPage){
		$("#inp_fm_page").val(curPage);
		var wrapId = $("#inp_fm_json_wrap").val();
		if(wrapId != ""){
			var url = $("#fm_page_req").attr("action");
			var data = {};
			data.inp_fm_page${5} = curPage;
			data.inp_fm_pageMax${5} = ${4};
			var jquWrap = $("#"+wrapId);
			var callBack = function(data){
				jquWrap.html(data);
			};
			$.post(url,data,callBack);
		}else{
			$("#fm_page_req").submit();
		}
	};
	function click2PrePage(){
		var curPage = $("#inp_fm_page").val();
		curPage--;
		if(curPage <= 0)
			curPage = 1;
		click2sendPaging(curPage);
	};
	
	function click2NxtPage(){
		var curPage = $("#inp_fm_page").val();
		curPage++;
		var pageMax = $("#inp_fm_pageMax").val();
		if(curPage > pageMax)
			curPage = pageMax;
		click2sendPaging(curPage);
	};
</script>
